## Part 3 of Problem 1 – unemployment data.

``` r
unemployment_long <- unemployment %>% 
  pivot_longer(
    cols = -Year,          
    names_to = "month",
    values_to = "unemployment_rate"
  ) %>% 
  rename(year = Year) %>%  
  mutate(
    month = case_when(
      month %in% c("1", "01", "Jan") ~ "January",
      month %in% c("2","02","Feb") ~ "February",
      month %in% c("3","03","Mar") ~ "March",
      month %in% c("4","04","Apr") ~ "April",
```
